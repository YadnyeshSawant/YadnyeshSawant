function countUp(el, target, suffix) {
  const dur = 900;
  const start = performance.now();
  function tick(now) {
    const p = Math.min(1, (now - start) / dur);
    const val = Math.round(target * (1 - Math.pow(1 - p, 3)));
    el.textContent = val.toLocaleString() + (suffix || "");
    if (p < 1) requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

function levelFor(count) {
  if (count === 0) return 0;
  if (count <= 5) return 1;
  if (count <= 15) return 2;
  if (count <= 30) return 3;
  if (count <= 50) return 4;
  return 5;
}
const PALETTE = ["#1c2128", "#0e4429", "#146c37", "#26a648", "#39d353", "#7cf2a0"];

function buildHeatmapEl() {
  const wrap = document.createElement("div");
  wrap.className = "heatmap-wrap";
  const grid = document.createElement("div");
  grid.className = "heatmap";

  const days = DATA.days;
  const first = new Date(days[0].date + "T00:00:00");
  const leadPad = (first.getDay()); // sunday=0 already
  let col = new Array(leadPad).fill(null);

  const cols = [];
  days.forEach((d) => {
    const date = new Date(d.date + "T00:00:00");
    const weekday = date.getDay();
    while (col.length < weekday) col.push(null);
    col.push(d);
    if (col.length === 7) { cols.push(col); col = []; }
  });
  if (col.length) { while (col.length < 7) col.push(null); cols.push(col); }

  cols.forEach((column) => {
    column.forEach((d) => {
      const cell = document.createElement("div");
      cell.className = "cell";
      if (!d) {
        cell.style.background = "transparent";
        cell.style.cursor = "default";
      } else {
        const lvl = levelFor(d.count);
        cell.style.background = PALETTE[lvl];
        cell.addEventListener("mousemove", (e) => {
          tooltip.style.display = "block";
          tooltip.style.left = e.clientX + 14 + "px";
          tooltip.style.top = e.clientY + 14 + "px";
          tooltip.textContent = `${d.date}: ${d.count} contribution${d.count === 1 ? "" : "s"}`;
        });
        cell.addEventListener("mouseleave", () => { tooltip.style.display = "none"; });
      }
      grid.appendChild(cell);
    });
  });

  wrap.appendChild(grid);

  const legend = document.createElement("div");
  legend.className = "legend";
  legend.innerHTML = "less ";
  PALETTE.forEach((c) => {
    legend.innerHTML += `<span class="cell" style="background:${c}"></span> `;
  });
  legend.innerHTML += " more";
  wrap.appendChild(legend);

  return wrap;
}

function runCommand(raw) {
  const trimmed = raw.trim();
  line(`<span class="prompt-tag">yadnyesh@github:~$</span> <span class="out-white">${escapeHtml(trimmed)}</span>`, "prompt-line");
  if (!trimmed) return;

  if (pendingAction === "resume") {
    const choice = trimmed.toLowerCase();
    pendingAction = null;
    if (choice === "y" || choice === "yes") {
      line(`<span class="out-green">opening resume in a new tab...</span>`);
      window.open("./Yadnyesh_Resume.pdf", "_blank");
    } else {
      line(`<span class="out-dim">cancelled.</span>`);
    }
    return;
  }

  const [cmd, ...args] = trimmed.split(/\s+/);
  const key = cmd.toLowerCase();
  if (key === "open") {
    const target = (args[0] || "").toLowerCase();
    if (target === "resume") {
      COMMANDS.resume();
      return;
    }
    const map = {
      github: "https://github.com/YadnyeshSawant",
      linkedin: "https://linkedin.com/in/yadnyesh-sawant",
      leetcode: "https://leetcode.com/yadnyesh",
    };
    if (map[target]) { line(`<span class="out-dim">opening ${target}...</span>`); window.open(map[target], "_blank"); }
    else line(`<span class="out-dim">usage: open [resume|github|linkedin|leetcode]</span>`);
    return;
  }
  if (COMMANDS[key]) { COMMANDS[key](args); }
  else line(`<span class="out-dim">command not found: ${escapeHtml(cmd)} — type 'help'</span>`);
}

function escapeHtml(s) {
  return s.replace(/[&<>"']/g, (c) => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#39;" }[c]));
}

let matrixOn = false;
function toggleMatrix() {
  matrixOn = !matrixOn;

  canvas.style.display = matrixOn ? "block" : "none";
  if (matrixOn) startMatrix();

function startMatrix() {
  const canvas = document.getElementById("matrixCanvas");
  const ctx = canvas.getContext && canvas.getContext("2d");
  if (!ctx) return; // canvas 2d not available in this environment -- skip quietly
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
  const cols = Math.floor(canvas.width / 14);
  const drops = new Array(cols).fill(0);
  const chars = "01YADNYESHSAWANT<>/{}[]#*";
  function draw() {
    if (!matrixOn) return;
    ctx.fillStyle = "rgba(13,17,23,0.12)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = "#39d353";
    ctx.font = "13px monospace";
    drops.forEach((y, i) => {
      const ch = chars[Math.floor(Math.random() * chars.length)];
      ctx.fillText(ch, i * 14, y * 14);
      drops[i] = y * 14 > canvas.height && Math.random() > 0.975 ? 0 : y + 1;
    });
    requestAnimationFrame(draw);
  }
  draw();
}

function countUp(el, target, suffix) {
  const dur = 900;
  const start = performance.now();
  function tick(now) {
    const p = Math.min(1, (now - start) / dur);
    const val = Math.round(target * (1 - Math.pow(1 - p, 3)));
    el.textContent = val.toLocaleString() + (suffix || "");
    if (p < 1) requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

function renderStats() {
  if (!DATA) return;
  const box = document.getElementById("statsBox");
  if (!box) return;
  const stats = [
    ["total", DATA.total_contributions, "contributions"],
    ["current", DATA.current_streak.length, "day streak"],
    ["longest", DATA.longest_streak.length, "day best streak"],
    ["best", DATA.best_day.count, "in one day"],
  ];
  box.innerHTML = "";
  stats.forEach(([key, val, label]) => {
    const card = document.createElement("div");
    card.className = "stat";
    card.innerHTML = `<div class="num" data-key="${key}">0</div><div class="lbl">${label}</div>`;
    box.appendChild(card);
    countUp(card.querySelector(".num"), val);
  });
}

function levelFor(count) {
  if (count === 0) return 0;
  if (count <= 5) return 1;
  if (count <= 15) return 2;
  if (count <= 30) return 3;
  if (count <= 50) return 4;
  return 5;
}
const PALETTE = ["#1c2128", "#0e4429", "#146c37", "#26a648", "#39d353", "#7cf2a0"];

function buildHeatmapEl() {
  const wrap = document.createElement("div");
  wrap.className = "heatmap-wrap";
  const grid = document.createElement("div");
  grid.className = "heatmap";

  const days = DATA.days;
  const first = new Date(days[0].date + "T00:00:00");
  const leadPad = (first.getDay()); // sunday=0 already
  let col = new Array(leadPad).fill(null);

  const cols = [];
  days.forEach((d) => {
    const date = new Date(d.date + "T00:00:00");
    const weekday = date.getDay();
    while (col.length < weekday) col.push(null);
    col.push(d);
    if (col.length === 7) { cols.push(col); col = []; }
  });
  if (col.length) { while (col.length < 7) col.push(null); cols.push(col); }

  cols.forEach((column) => {
    column.forEach((d) => {
      const cell = document.createElement("div");
      cell.className = "cell";
      if (!d) {
        cell.style.background = "transparent";
        cell.style.cursor = "default";
      } else {
        const lvl = levelFor(d.count);
        cell.style.background = PALETTE[lvl];
        cell.addEventListener("mousemove", (e) => {
          tooltip.style.display = "block";
          tooltip.style.left = e.clientX + 14 + "px";
          tooltip.style.top = e.clientY + 14 + "px";
          tooltip.textContent = `${d.date}: ${d.count} contribution${d.count === 1 ? "" : "s"}`;
        });
        cell.addEventListener("mouseleave", () => { tooltip.style.display = "none"; });
      }
      grid.appendChild(cell);
    });
  });

  wrap.appendChild(grid);

  const legend = document.createElement("div");
  legend.className = "legend";
  legend.innerHTML = "less ";
  PALETTE.forEach((c) => {
    legend.innerHTML += `<span class="cell" style="background:${c}"></span> `;
  });
  legend.innerHTML += " more";
  wrap.appendChild(legend);

  return wrap;
}

let pendingAction = null;

function runCommand(raw) {
  const trimmed = raw.trim();
  line(`<span class="prompt-tag">yadnyesh@github:~$</span> <span class="out-white">${escapeHtml(trimmed)}</span>`, "prompt-line");
  if (!trimmed) return;

  if (pendingAction === "resume") {
    const choice = trimmed.toLowerCase();
    pendingAction = null;
    if (choice === "y" || choice === "yes") {
      line(`<span class="out-green">opening resume in a new tab...</span>`);
      window.open("./Yadnyesh_Resume.pdf", "_blank");
    } else {
      line(`<span class="out-dim">cancelled.</span>`);
    }
    return;
  }

  const [cmd, ...args] = trimmed.split(/\s+/);
  const key = cmd.toLowerCase();
  if (key === "open") {
    const target = (args[0] || "").toLowerCase();
    if (target === "resume") {
      COMMANDS.resume();
      return;
    }
    const map = {
      github: "https://github.com/YadnyeshSawant",
      linkedin: "https://linkedin.com/in/yadnyesh-sawant",
      leetcode: "https://leetcode.com/yadnyesh",
    };
    if (map[target]) { line(`<span class="out-dim">opening ${target}...</span>`); window.open(map[target], "_blank"); }
    else line(`<span class="out-dim">usage: open [resume|github|linkedin|leetcode]</span>`);
    return;
  }
  if (COMMANDS[key]) { COMMANDS[key](args); }
  else line(`<span class="out-dim">command not found: ${escapeHtml(cmd)} — type 'help'</span>`);
}

function escapeHtml(s) {
  return s.replace(/[&<>"']/g, (c) => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#39;" }[c]));
}

    const val = input.value;
    input.value = "";
    runCommand(val);
  }

let matrixOn = false;
function toggleMatrix() {
  matrixOn = !matrixOn;
  const canvas = document.getElementById("matrixCanvas");
  canvas.style.display = matrixOn ? "block" : "none";
  if (matrixOn) startMatrix();
}
function startMatrix() {
  const canvas = document.getElementById("matrixCanvas");
  const ctx = canvas.getContext && canvas.getContext("2d");
  if (!ctx) return; // canvas 2d not available in this environment -- skip quietly
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
  const cols = Math.floor(canvas.width / 14);
  const drops = new Array(cols).fill(0);
  const chars = "01YADNYESHSAWANT<>/{}[]#*";
  function draw() {
    if (!matrixOn) return;
    ctx.fillStyle = "rgba(13,17,23,0.12)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = "#39d353";
    ctx.font = "13px monospace";
    drops.forEach((y, i) => {
      const ch = chars[Math.floor(Math.random() * chars.length)];
      ctx.fillText(ch, i * 14, y * 14);
      drops[i] = y * 14 > canvas.height && Math.random() > 0.975 ? 0 : y + 1;
    });
    requestAnimationFrame(draw);
  }
  draw();
}
